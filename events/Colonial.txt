			}

### Events to give persecuted minorities into colonies ###

province_event = {
	id = colonial.5
	title = "EVTNAME1423"
	desc = "EVTDESC1423"
	picture = STREET_SPEECH_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		is_overseas = yes
		is_colony = yes
		colonysize = 600
		religion = owner
		NOT = { has_province_modifier = religious_migration }
		owner = {
			NOT = {
				any_owned_province = {
					expulsion_target = {
						province_id = root
					}
				}
			}
			any_owned_province = {
				is_overseas = no
				religion_group = ROOT
				NOT = { religion = ROOT }
				expelling_minorities = no
			}
		}
	}
	
	immediate = {
		hidden_effect = {
			owner = {
				random_owned_province = {
					limit = {
						is_overseas = no
						religion_group = ROOT
						NOT = { religion = ROOT }
						expelling_minorities = no
					}
					save_event_target_as = heretics_from_here
				}
			}
		}
	}

	mean_time_to_happen = {
		days = 1
	}

	option = {
		name = "EVTOPTA1423"
		goto = ROOT
		change_religion = event_target:heretics_from_here
		change_culture = owner
		add_colonysize = 400
		add_province_modifier = {
			name = "religious_migration"
			duration = 8760
		}
	}
	option = {
		name = "colonial.5.b"
		goto = heretics_from_here
		event_target:heretics_from_here = {
			add_province_modifier = {
				name = "enforcing_unity"
				duration = 3650
			}
		}
		add_colonysize = -100
	}
}

# Fur Trade Dwindles
province_event = {
	id = colonial.13
	title = "colonial.EVTNAME13"
	desc = "colonial.EVTDESC13"
	picture = MERCHANTS_TALKING_eventPicture
	goto = ROOT
	
	is_triggered_only = yes
	
	trigger = {
		trade_goods = fur
		OR = {
			NOT = { has_province_flag = received_trade_goods }
			had_province_flag = { flag = received_trade_goods days = 3650 }
		}
		NOT = { has_empty_adjacent_province = yes }
		NOT = {
			any_neighbor_province = {
				owner = {
					has_reform = steppe_horde
				}
				controller = {
					has_reform = steppe_horde
				}
			}
		}
	}

	immediate = {
		hidden_effect = {
			clr_province_flag = received_trade_goods
			set_province_flag = received_trade_goods
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = "colonial.EVTOPTA13"
		change_trade_goods = naval_supplies
	}
	option = {
		name = "colonial.EVTOPTC13"
		change_trade_goods = grain
	}
}

# Colonial Migration
province_event = {
	id = colonial.17
	title = "EVTNAME6401"
	desc = "EVTDESC6401"
	picture = COLONIZATION_eventPicture
	goto = ROOT

	is_triggered_only = yes

	trigger = {
		colonysize = 300
		NOT = { colonysize = 500 }
		owner = {
			any_owned_province = {
				is_colony = yes
				colonysize = 500
				NOT = { province_id = ROOT }
			}
		}
	}
	
	immediate = {
		hidden_effect = {
			owner = {
				random_owned_province = {
					limit = {
						is_colony = yes
						colonysize = 500
						NOT = { province_id = ROOT }
					}
					save_event_target_as = migration_province_target
				}
			}
		}
	}

	mean_time_to_happen = {
		days = 1
	}

	option = {
		name = "EVTOPTA6401"
		goto = migration_province_target
		add_colonysize = -200
		event_target:migration_province_target = {
			add_colonysize = 200
		}
	}
}
