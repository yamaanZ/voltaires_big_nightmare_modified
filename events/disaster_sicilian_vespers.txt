country_event = {
	id = sicilian_vespers.7
	title = sicilian_vespers.7.t
	desc = sicilian_vespers.7.d
	picture = POPE_PREACHING_eventPicture

	fire_only_once = yes

	mean_time_to_happen = {
		months = 12
	}

	trigger = {
		tag = PAP
		TNC = {
			exists = yes
			has_country_modifier = vespers_victorious
		}
	}

	option = {
		name = sicilian_vespers.7.a	
		ai_chance = { factor = 50 }
		TNC = {
			add_country_modifier = {
				name = "vespers_papal_recognition"
				duration = 9125
			}
		}
		hidden_effect = {
			every_known_country = {
				limit = {
					uses_papacy = yes
				}
				if = {
					limit = {
						sicilia_region = {
							OR = {
								is_core = PREV
								is_permanent_claim = PREV
							}
						}
					}
					sicilia_region = {
						limit = {
							OR = {
								is_core = ROOT
								is_permanent_claim = ROOT
							}
						}
						remove_core = PREV
						remove_claim = PREV
					}
				}
			}
		}
	}
	option = {
		name = sicilian_vespers.7.b	
		ai_chance = { factor = 50 }
		ARA = {
			if = {
				limit = {
					exists = yes
					num_of_cities = 18
					NOT = { has_country_flag = def_by_vespers }
				}
				set_country_flag = claim_sic_ara
				country_event = {
					id = sicilian_vespers.8
					days = 30
					random = 10
				}
			}
		}
		random_known_country = {
			limit = { has_country_flag = def_by_vespers }
			create_union = TNC
			random_rival_country = {
				limit = {
					OR = {
						religion = catholic 
						religion = catholique
					}
					government = monarchy
					capital_scope = {
						OR = {
							superregion = french_superregion
							region = lotharingia_region
							superregion = iberia_superregion
							superregion = italy_superregion
							region = south_italy_region
							superregion = balkans_superregion
							superregion = africa_superregion
						}
					}
					NOT = {
						OR = {
							ruler_has_personality = malevolent_personality
							ruler_has_personality = greedy_personality
							ruler_has_personality = cruel_personality
						}
					}
				}
				if = {
					limit = {
						NOT = {
							ARA = { has_country_flag = claim_sic_ara }
						}
					}
					set_country_flag = claim_sic_ara
					country_event = {
						id = sicilian_vespers.8
						days = 30
						random = 10
					}
				}
			}
		}
		TNC = {
			add_liberty_desire = 100
			overlord = {
				reverse_add_opinion = {
					who = TNC
					modifier = "vespers_tyrant_rule"
				}
			}
		}
	}
}

	}

		trigger = {
			stability = 1

	}

			}

						is_core = ROOT

			}

					kill_units = {
						who = owner
					}
				}
			}
		}
		TNC = {
			if = {
				limit = {
					2041 = {
						religion = catholique
					}
				}
				change_religion = catholique
			}

				limit = {

						religion = catholic
					}
				}
				change_religion = catholic
			}
			if = {
				limit = {
					2041 = {
						OR = {
							AND = {
								religion_group = christian
								NOT = { religion = catholic }
								NOT = { religion = catholique }
							}
							religion_group = muslim 
						}
					}
				}
				change_religion = PREV
				2041 = {
					change_religion = PREV
				}
			}
			add_country_modifier = {
				name = "vespers_victorious"
				duration = -1
			}
		}
		hidden_effect = {
			set_country_flag = def_by_vespers
		}
		add_prestige = -25
		add_stability = -1

			limit = {
				NOT = { capital = 564 }
				owns = 564
			}
