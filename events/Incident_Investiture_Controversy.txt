# event for owner of milan to start the incident
country_event = {
	id = incident_investiture_controversy.1
	title = incident_investiture_controversy.1.t
	desc = incident_investiture_controversy.1.d
	picture = BAD_POPE_eventPicture
	
	major = yes
	fire_only_once = yes
	
	trigger = {
		has_dlc = "Emperor"
		normal_or_historical_nations = yes
		NOT = { started_in = 1070.1.1 }
		always = no
		NOT = { active_imperial_incident = any }
		NOT = { has_reform = papacy_reform }
		owns = 1818
		religion = catholic
		papacy_active = yes
		is_part_of_hre = yes
	}

	mean_time_to_happen = {
		months = 120
	}
	
	option = {
		name = incident_investiture_controversy.1.a
		add_stability = -1
		if = {
			limit = { government = theocracy }
			tooltip = { kill_ruler = yes }
			define_ruler = {
				change_adm = -1
				change_dip = -1
				change_mil = -1
			}
		}
		if = {
			limit = { has_estate = estate_church }
			add_estate_influence_modifier = {
				estate = estate_church
				desc = EST_VAL_PATARIA_COUP
				influence = 15
				duration = 5475
			}
		}
		else = {
			1818 = {
				catholic_rebels = 2
			}
		}
		set_imperial_incident = incident_investiture_controversy
	}
}

	picture = BAD_POPE_eventPicture

	trigger = {
		normal_or_historical_nations = yes
		always = no
		had_global_flag = {
			flag = investiture_controversy_active
			days = 3650
		}
		is_emperor = yes
		religion = catholic
		papacy_active = yes
		NOT = { war_with = PAP }
		NOT = { truce_with = PAP }
		PAP = { is_subject = no }
	}

	mean_time_to_happen = {
		months = 120
	}

	option = {
		name = "incident_investiture_controversy.6.a"
		ai_chance = {
			factor = 40
			modifier = {
				factor = 2
				reverse_has_opinion = {
					who = PAP
					value = 100
				}
			}
		}
		custom_tooltip = incident_investiture_controversy.6.tt
		add_years_of_income = -0.5
		set_ruler_flag = excommunicated_emperor
		add_ruler_modifier = {
			name = investiture_excommunication_m
			duration = -1
		}
	}
	option = {
		name = "incident_investiture_controversy.6.b"
		ai_chance = {
			factor = 40
			modifier = {
				factor = 0
				OR = {
					has_any_disaster = yes
					NOT = { manpower_percentage = 0.5 }
				}
			}
			modifier = {
				factor = 2
				army_size = PAP
			}
		}
		add_prestige = 20
		declare_war_with_cb = {
			casus_belli = cb_imperial_supremacy
			who = PAP
		}
		add_ruler_modifier = {
			name = investiture_excommunication_m
			duration = -1
		}
		if = {
			limit = { ai = yes }
			every_ally = {
				limit = {
					ai = yes
					is_at_war = no
					has_any_disaster = no
					NOT = { average_unrest = 1 }
				}
				join_all_offensive_wars_of = ROOT
			}
		}
	}
	option = {
		name = "incident_investiture_controversy.6.c"
		ai_chance = {
			factor = 20
		}
		add_legitimacy = -10
		add_ruler_modifier = {
			name = investiture_excommunication_m
			duration = -1
		}
	}
}

		}
	}

	}

	option = {
		name = "incident_investiture_controversy.7.a"

				factor = 1.5

					who = emperor

				}
			}
			modifier = {
				factor = 1.5
				has_opinion = {
					who = emperor
					value = 150
				}
			}
		}
		emperor = {
			set_country_flag = excommunication_is_lifted

				remove_country_modifier = investiture_excommunication_m
			}
			hidden_effect = {
				country_event = {
					id = incident_investiture_controversy.8
				}
			}
		}
	}
	option = {
		name = "incident_investiture_controversy.7.b"
		ai_chance = {
			factor = 50
			modifier = {
				factor = 2
				NOT = {
					has_opinion = {
						who = emperor
						value = 0
					}
				}
			}
			modifier = {
				factor = 2
				NOT = {
					has_opinion = {
						who = emperor
						value = -100
					}
				}
			}
		}

			set_country_flag = excommunication_is_not_lifted

				}
			}
		}
		add_prestige = 10
	}
