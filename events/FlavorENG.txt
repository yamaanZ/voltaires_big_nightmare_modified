# The Lollard Heresy
country_event = {
	id = flavor_eng.9100
	title = "flavor_eng.EVTNAME9100"
	desc = "flavor_eng.EVTDESC9100"
	picture = {	
		trigger = {
			has_dlc = "Emperor"
		}
		picture = ADULT_BAPTISME_eventPicture
	}
	picture = {	
		trigger = {
			NOT = { has_dlc = "Emperor" }
		}
		picture = RELIGION_eventPicture
	}

	fire_only_once = yes
	major = yes

	trigger = {
		tag = ENG
		is_year = 1375
		NOT = { has_global_flag = lollard_heresy_g }
		num_of_cities = 15
		OR = {
			religion = catholic
			religion = catholique
		}
		papacy_active = yes
		NOT = { any_province = { religion = lollardy } }
		any_province = {
			superregion = british_isles_superregion
			owned_by = ENG
			can_have_center_of_reformation_trigger = {
				RELIGION = lollardy
			}
		}
	}

	immediate = {
		hidden_effect = {
			set_global_flag = lollard_heresy_g
			enable_religion = lollardy
		}
	}
	
	mean_time_to_happen = { months = 120 }

	option = {		# Stop the Heretics
		name = "flavor_eng.EVTOPTA9100"
		ai_chance = { factor = 75 }
		random_owned_province = {
			limit = {
				superregion = british_isles_superregion
				can_have_center_of_reformation_trigger = {
					RELIGION = lollardy
				}
			}
			change_religion = lollardy
			add_reform_center = lollardy
			add_province_modifier = {
				name = lollards
				duration = 9000
			}
		}
		add_country_modifier = {
			name = eng_opposes_lollards
			duration = -1
		}	
		if = {
			limit = { religion = catholic }
			PAP = {
				add_opinion = {
					who = ROOT
					modifier = eng_opposes_lollards
				}
			}
		}	
		if = {
			limit = { religion = catholique }
			AVI = {
				add_opinion = {
					who = ROOT
					modifier = eng_opposes_lollards
				}
			}
		}		
	}
	option = {		# Let them be
		name = "flavor_eng.EVTOPTB9100"
		ai_chance = { factor = 25 }
		random_owned_province = {
			limit = {
				superregion = british_isles_superregion
				can_have_center_of_reformation_trigger = {
					RELIGION = lollardy
				}
			}
			change_religion = lollardy
			add_reform_center = lollardy
			add_permanent_province_modifier = {
				name = "religious_zeal_at_conv"
				duration = 9000
			}
		}
		add_country_modifier = {
			name = eng_tolerates_lollards
			duration = -1
		}
		if = {
			limit = { religion = catholic }
			PAP = {
				add_opinion = {
					who = ROOT
					modifier = eng_tolerates_lollards
				}
			}
		}
		if = {
			limit = { religion = catholique }
			AVI = {
				add_opinion = {
					who = ROOT
					modifier = eng_tolerates_lollards
				}
			}
		}
		if = {
			limit = {
				FRA = {
					exists = yes
					OR = {
						religion = catholic
						religion = catholique
					}
				}
			}
			FRA ={
				add_opinion = {
					who = ROOT
					modifier = eng_tolerates_lollards
				}
			}
		}
		if = {

# The End of the Hundred Years War
country_event = {
	id = flavor_eng.9104
	title = "flavor_eng.EVTNAME9104"
	desc = "flavor_eng.EVTDESC9104"
	picture = DIPLOMACY_eventPicture
	
	fire_only_once = yes

	trigger = {
		NOT = { is_year = 1500 }
		NOT = { has_global_flag = eng_had_end_of_hundred_years_war }
		had_global_flag = { flag = eng_had_start_of_hundred_years_war days = 18250 }
		OR = {
			tag = ENG
			AND = {
				tag = GBR
				NOT = { exists = ENG }
			}
		}
		primary_culture = english
		NOT = { has_reform = feudal_france_reform }
		NOT = {
			french_superregion = { owned_by = ROOT }
			lotharingia_region = { owned_by = ROOT }
		}
		any_country = {
			has_reform = feudal_france_reform
			NOT = { is_subject_of = ENG }
			NOT = { is_subject_of = GBR }
			NOT = { war_with = ENG }
			NOT = { war_with = GBR }
		}
	}
	
	immediate = {
		hidden_effect = {
			set_global_flag = eng_had_end_of_hundred_years_war
			if = {
				limit = {
					NOT = { any_country = { has_country_flag = had_end_of_HYW_france } }
				}
				random_country = {
					limit = { has_reform = feudal_france_reform }
					country_event = { id = flavor_fra.9104 days = 1 }
				}
			}
		}
	}
	
	mean_time_to_happen = {
		months = 12
	}
	
	option = {			# We have lost our foothold in France
		name = "flavor_eng.EVTOPTA9104"
		ai_chance = { factor = 80 }
		add_stability_or_adm_power = yes
		every_core_province = {
			limit = {
				OR = {
					superregion = french_superregion
					region = lotharingia_region
				}
				NOT = { owned_by = ROOT }
			}
			remove_core = ROOT
		}
		random_country = {
			limit = { has_reform = feudal_france_reform }
			add_opinion = {
				who = ROOT
				modifier = eng_no_english_in_france
			}
		}
	}
	option = {		# We will never give this up!
		name = "flavor_eng.EVTOPTB9104"
		ai_chance = { factor = 20 }
		add_stability = -1
		random_country = {
			limit = { has_reform = feudal_france_reform }
			add_opinion = {
				who = ROOT
				modifier = eng_refuse_to_give_up_france
			}
		}
	}
}
