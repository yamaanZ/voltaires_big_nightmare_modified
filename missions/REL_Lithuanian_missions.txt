				if = {

						any_ally = {
							has_opinion = {
								who = ROOT

							}

								culture = latvian
								culture = selonian
								culture = semigallian

					add_prestige_or_monarch_power = { amount = 10 } 
				}

					add_mil_power = 100
                	add_army_tradition = 10
				}
			}
