vn_crn_missions_slot_2 = {
    slot = 7
    generic = no

    potential = {
        tag = CRN
        NOT = { map_setup = map_setup_random }
    }
    has_country_shield = yes

    vn_crn_brothers_in_brittany = {
        icon = mission_rb_conquer_brittany
        position = 1
        required_missions = {  }
        trigger = {
            if = {
                limit = {
                    any_country = { primary_culture = breton }
                }
                any_country = {
                    primary_culture = breton
                    has_opinion = {
                        who = ROOT
                        value = 125
                    }
                }
            }
        }
        effect = {
            if = {
                limit = { any_country = { primary_culture = breton } }
                random_country = {
                    limit = {
                        primary_culture = breton 
                        has_opinion = {
                            who = ROOT
                            value = 125
                        }
                    }
                    add_opinion = {
                        who = ROOT
                        modifier = crn_diplomacy_opinion
                    }
                }
            }
            else = {
                custom_tooltip = vn_bypass_this_mission  #Bypass
            }
            custom_tooltip = vn_crn_breton_diplomacy.tt
            hidden_effect = {
                every_country = {
                    limit = { 
                        is_subject = no
                        primary_culture = breton 
                        has_opinion = {
                            who = ROOT
                            value = 125
                        }
                    }
                    random = {
                        chance = 50
                        every_owned_province = {
                            limit = { 
                                OR = {
                                    area = raozhon_area
                                    area = nantes_area
                                    area = finistere_area
                                }
                                NOT = { is_core = ROOT }
                                NOT = { is_permanent_claim = ROOT }
                            }
                            add_permanent_claim = ROOT
                        }
                    }
                }
            }
        }
    }
