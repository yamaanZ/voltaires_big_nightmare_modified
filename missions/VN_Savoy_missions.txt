        icon = knights_of_the_new_world
        required_missions = { vn_sav_fortify_chambery }

			if = {
				limit = { religion = catholic exists = PAP }
				OR = {
					AND = {
						reverse_has_opinion = {
							who = PAP
							value = 100
						}
						papal_influence = 25
					}
					is_rival = PAP
                    NOT = { exists = PAP }
				}
			}
			else_if = {
				limit = { religion = catholique exists = AVI }
				OR = {
					AND = {
						reverse_has_opinion = {
							who = AVI
							value = 100
						}
						papal_influence = 25
					}
					is_rival = AVI
                    NOT = { exists = AVI }
				}
			}
			else = {
				grown_by_development = 100
			}
            calc_true_if = {
                amount = 2
                adm = 4
                dip = 4
                mil = 4
            }
        }
        effect = {
            add_religious_influence_effect_large = yes
            add_country_modifier = {
                name = sav_order_of_saint_maurice_mission
                duration = -1
            }
        }

        icon = become_king
        required_missions = { vn_sav_conquer_montferrat vn_sav_conquer_aosta vn_sav_conquer_savoy }
        trigger = {
            if = {
                limit = { hre_size = 1 }
                OR = {
                    is_part_of_hre = no
                    is_emperor = yes
                    reverse_has_opinion = {
                        who = emperor
                        value = 150
                    }
                }
            }
            else = {
                government_rank = 4
            }
        }
        effect = {
            if = {
                limit = { NOT = { government_rank = 3 } }
                set_government_rank = 3
            }
            else = {
                add_stability_or_adm_power = yes
            }
            add_prestige_or_monarch_power = { amount = 15 } 
            add_legitimacy = 10
        }
    }
    vn_sav_defy_france = {
        icon = unite_the_wittelsbachs
        required_missions = { vn_sav_become_a_duchy }
        provinces_to_highlight = {
            if = {
                limit = { exists = FRA }
                is_capital_of = FRA
            }
            else = {
                always = no
            }
        }
        trigger = {
            OR = {
                FRA = {
                    custom_trigger_tooltip = {
						tooltip = has_been_insulted_by_root
						OR = {
							has_opinion_modifier = {
								who = ROOT
								modifier = insulted
							}
							has_opinion_modifier = {
								who = ROOT
								modifier = scornfully_insulted
							}
						}
					}
                }
                NOT = { exists = FRA }
            }
            total_development = 200
        }
        effect = {
            add_country_modifier = {
                name = sav_defy_france_mission
                duration = 9125
            }
        }
    }
