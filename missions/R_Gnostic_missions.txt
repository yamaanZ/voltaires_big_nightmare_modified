vn_gnostic_missions = {
	slot = 3
	generic = yes
    ai = yes
	potential = {
		religion_group = zoroastrian_group
		NOT = { religion = zoroastrian }
	}

	vn_fortresses_of_faith = {
		icon = mission_control_surat
		required_missions = { }
		position = 9
		trigger = {
			if = {
				limit = { has_estate = estate_church }
				num_of_owned_provinces_with = {
					value = 3
					has_fort_trigger = yes
					NOT = { unrest = 0.01 }
					has_owner_religion = yes
				}
			}
			else = {
				num_of_owned_provinces_with = {
					value = 3
					has_fort_trigger = yes
					has_owner_religion = yes
				}
				religious_unity = 1
			}
		}
		effect = {
			add_country_modifier = {
				name = "fortresses_of_faith_mission"
				duration = 9125
			}
		}
	}
	vn_written_tradition = {
		icon = mission_court_education
		required_missions = { vn_fortresses_of_faith }
		trigger = {
			has_institution = printing_press
			all_state_province = {
				printing_press = 100
			}
			capital_scope = {
				development = 12
			}
		}
		effect = {
			add_country_modifier = {
				name = "written_tradition_mission"
				duration = 9125
			}
		}
	}
	vn_having_knowledge = {
		icon = mission_friends_or_enemies
		required_missions = { }
		trigger = {
			if = {
				limit = { has_dlc = "Rule Britannia" }
				innovativeness = 25
			}
			num_of_owned_provinces_with = {
				value = 5
				OR = {
					has_building = college
					has_education_building_trigger = yes
				}
			}
		}
		effect = {
			add_country_modifier = {
				name = "having_knowledge_mission"
				duration = 9125
			}
		}
	}
}
