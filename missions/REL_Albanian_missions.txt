VN_albanian_missions = {
	slot = 5
	generic = yes
	ai = yes
	potential = {
		culture_group = albanian_group
		NOT = { map_setup = map_setup_random }
	}
	
	vn_albanian_defence = {
		icon = mission_conquer_50_development
        required_missions = {}
        position = 5
		trigger = {
			if = {
				limit = { has_dlc = "Cradle of Civilization" }
				num_of_regiments_at_full_drill = 10
			}
			else = {
				army_tradition = 25
			}
			capital_scope = {
				has_fort_trigger = yes
			}
		}
		effect = {
			create_general = {
				tradition = 75
			}
			add_country_modifier = {
				name = "albanian_defence_mission"
				duration = 9125
			}
		}
	}
	vn_abolish_albanian_government = {
		icon = mission_coronation
		required_missions = { vn_albanian_defence }
		trigger = {
            if = {
                limit = { has_dlc = "Common Sense" }
                legitimacy_equivalent = 100
            }
            else_if = {
                limit = { government = monarchy }
                legitimacy = 100
            }
			is_tribal = no
		}
		effect = {
			add_country_modifier = {
				name = "abolish_albanian_government_mission"
				duration = 9125
			}
		}
	}
	vn_trade_of_ragusa = {
		icon = mission_dominate_home_trade_node
		required_missions = { vn_abolish_albanian_government }
		provinces_to_highlight = {
			province_id = 137
		}
		trigger = {
			137 = {
				trade_share = {
					country = ROOT
					share = 25
				}
			}
		}
		effect = {
			add_country_modifier = {
				name = "trade_of_ragusa_mission"
				duration = 9125
			}
		}
	}
	vn_athleta_christi = {
		icon = mission_crusade_for_varna
		required_missions = { vn_trade_of_ragusa }
		trigger = {
			is_defender_of_faith = yes
			num_of_cities = 25
		}
		effect = {
			add_prestige_or_monarch_power = { amount = 10 } 
			add_country_modifier = {
				name = "athleta_christi_mission"
				duration = 9125
			}
		}
	}
}
