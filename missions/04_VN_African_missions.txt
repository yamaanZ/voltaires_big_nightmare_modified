vn_gen_africa_1 = {
	slot = 1
	generic = yes
	ai = yes
	potential = {
		capital_scope = {
			continent = africa
		}
		NOT = { has_reform = pirate_republic_reform }
	}
	
	africa_slave_trade = {
		position = 9
		icon = mission_african_soldier
		required_missions = { }
		trigger = {
			OR = {
				AND = {
					production_leader = {
						trade_goods = slaves
					}
					slaves = 4
				}
				AND = {
					has_country_modifier = the_abolish_slavery_act
					NOT = {
						any_neighbor_country = {
							slaves = 1
						}
					}
				}
			}
		}
		effect = {
			if = {
				limit = {
					production_leader = {
						trade_goods = slaves
					}
				}
				add_mercantilism = 2
				add_country_modifier = {
					name = "africa_slave_trade_mission"
					duration = 9125
				}
			}
			else = {
				add_country_modifier = {
					name = "african_abolitionists"
					duration = 9125
				}
			}
		}
	}
	africa_modernization = {
		icon = mission_have_manufactories
		required_missions = { }
		trigger = {
			is_tribal = no
			NOT = { has_reform = iqta }
			any_neighbor_country = {
				technology_group = western
				NOT = {
					tech_difference = 1
				}
			}
		}
		effect = {
			add_innovativeness_big_effect = yes
			add_prestige_or_monarch_power = { amount = 20 } 
		}
	}
	african_wealth_mission = {
		icon = mission_war_chest
		required_missions = { }
		trigger = {
			OR = {
				num_of_owned_provinces_with = {
					value = 3
					OR = {
						trade_goods = silver
						trade_goods = gold
					}
				}
				AND = {
					production_leader = {
						trade_goods = ivory
					}
					ivory = 7
				}
				production_leader = {
					trade_goods = coffee
				}
			}
		}
		effect = {
			add_country_modifier = {
				name = "wealth_of_africa_modifier"
				duration = 9125
			}
		}
	}
}
