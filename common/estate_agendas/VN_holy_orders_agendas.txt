					AND = {
						NOT = { has_building = customs_house }
						num_free_building_slots = 1
					}
					AND = {
						ROOT = { adm_tech = 21 }
						has_building = customs_house

				}
			}

			factor = 2
			NOT = {
				crown_land_share = 15
			}
		}
	}

		}

			limit = {
				is_state_core = root
				NOT = { has_building = arsenal }
				OR = {
					AND = {
						NOT = { has_building = customs_house }
						num_free_building_slots = 1
					}
					AND = {
						ROOT = { adm_tech = 21 }
						has_building = customs_house
					}

		random_owned_province = {
			limit = {
				is_state_core = root
				base_tax = 4
				NOT = { has_building = arsenal }

					AND = {
						NOT = { has_building = customs_house }

				is_state_core = root

					AND = {
						NOT = { has_building = customs_house }

					AND = {
						ROOT = { adm_tech = 21 }
						has_building = customs_house
					}
				}
			}
			save_event_target_as = agenda_province
		}
		event_target:agenda_province = {
			if = {
				limit = {
					NOT = { has_building = customs_house }
				}
				ROOT = {
					set_variable = {
						which = estate_holy_orders_build_customs_house_in_y_var
						value = 1
					}
				}
			}
			else = {
				ROOT = {
					set_variable = {
						which = estate_holy_orders_build_customs_house_in_y_var
						value = 2
					}
				}
			}

		province_id = event_target:agenda_province

		event_target:agenda_province = {
			NOT = { owned_by = root }
		}
	}

			limit = {

						has_building = customs_house

					}
				}
			}

		on_completed_agenda_effect = {
			estate = estate_holy_orders

		}

			limit = {
				NOT = { crown_land_share = 15 }

					territory = 10
				}
			}
			take_estate_land_share_small = { estate = estate_holy_orders }
		}
		else = {
			event_target:agenda_province = {
				add_base_tax = 1
			}
		}
    }
    failing_effect = {
		on_failed_agenda_effect = yes
		clr_auto_complete_flag = yes
		event_target:agenda_province = {
			if = {
				limit = { owned_by = root }
				add_named_unrest = {
					name = local_holy_orders_displeased
					value = 5
				}
			}
		}
		add_estate_loyalty_modifier = {
			estate = estate_holy_orders
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
		}
    }
}
