fragmentation_of_poland = {
	potential = {
		NOT = { has_global_flag = fragmentation_of_poland_ended }
		OR = {
			AND = {
				tag = POL
				NOT = { has_global_flag = had_fragmentation_of_poland }
			}
			AND = {
				tag = KRA
				has_global_flag = had_fragmentation_of_poland
			}
		}
		always = no
		is_emperor_of_china = no
		is_emperor = no
		is_free_or_tributary_trigger = yes
		normal_or_historical_nations = yes
		government = monarchy
	}

	can_start = {
		custom_trigger_tooltip = {
			tooltip = fragpol_tt.1
			NOT = { has_global_flag = had_fragmentation_of_poland }
		}
		has_any_disaster = no
		always = no
		is_year = 1100
		num_of_cities = 50
		calc_true_if = {
			SDZ = {
				OR = {
					exists = no
					is_subject_of = ROOT
				}
			}
			WKP = {
				OR = {
					exists = no
					is_subject_of = ROOT
				}
			}
			KSZ = {
				OR = {
					exists = no
					is_subject_of = ROOT
				}
			}
			KRA = {
				OR = {
					exists = no
					is_subject_of = ROOT
				}
			}
			SIR = {
				OR = {
					exists = no
					is_subject_of = ROOT
				}
			}
			MAZ = {
				OR = {
					exists = no
					is_subject_of = ROOT
				}
			}
			SIL = {
				OR = {
					exists = no
					is_subject_of = ROOT
				}
			}
			amount = 4
		}
	}
	
	can_stop = {
		OR = {
			has_any_disaster = yes
			NOT = { government = monarchy }
			NOT = { tag = POL }
			is_subject_other_than_tributary_trigger = yes
			always = no
		}
	}
		
	progress = {
		modifier = {
			factor = 0.5
			dynasty = "Piast"
		}
		modifier = {
			factor = 0.5
			has_heir = yes
		}
		modifier = {
			factor = 0.5
			NOT = { stability = 1 }
		}
		modifier = {
			factor = 0.5
			NOT = { stability = 0 }
		}
		modifier = {
			factor = 0.5
			NOT = { legitimacy_equivalent = 100 }
		}
		modifier = {
			factor = 0.5
			NOT = { legitimacy_equivalent = 75 }
		}
		modifier = {
			factor = 0.5
			NOT = { legitimacy_equivalent = 50 }
		}
		modifier = {
			factor = 0.5
			NOT = { legitimacy_equivalent = 25 }
		}
		modifier = {
			factor = 0.5
			NOT = {
				any_owned_province = {
					OR = {
						region = hinterpommern_region
						region = westpreussen_region
					}
				}
			}
		}
		modifier = {
			factor = 0.5
			NOT = { prestige = 0 }
		}
		modifier = {
			factor = 0.5
			NOT = { prestige = -50 }
		}
	}
	
	modifier = {
		legitimacy = -1
		diplomatic_reputation = -2
	}	
	
	can_end = {
		tag = POL
		custom_trigger_tooltip = {
			tooltip = pol_was_not_split		
			NOT = { has_global_flag = pol_was_split }
		}
		custom_trigger_tooltip = {
			tooltip = disaster_active_for_10_years_tooltip
			had_global_flag = {
				flag = poland_fragmentation_active
				days = 3650
			}
		}
		stability = 1
		is_at_war = no
		NOT = {
			any_owned_province = {
				has_rebel_faction = nationalist_rebels
			}
		}
		NOT = { num_of_rebel_controlled_provinces = 1 }
		calc_true_if = {
			MAZ = {
				OR = {
					exists = no
					is_subject = yes
					NOT = {
						any_owned_province = {
							superregion = poland_superregion
						}
					}
				}
			}
			SIL = {
				OR = {
					exists = no
					is_subject = yes
					NOT = {
						any_owned_province = {
							superregion = poland_superregion
						}
					}
				}
			}
			WKP = {
				OR = {
					exists = no
					is_subject = yes
					NOT = {
						any_owned_province = {
							superregion = poland_superregion
						}
					}
				}
			}
			SDZ = {
				OR = {
					exists = no
					is_subject = yes
					NOT = {
						any_owned_province = {
							superregion = poland_superregion
						}
					}
				}
			}
			KRA = {
				OR = {
					exists = no
					is_subject = yes
					NOT = {
						any_owned_province = {
							superregion = poland_superregion
						}
					}
				}
			}
			amount = 2
		}
	}
	
	on_start = fragmentation_of_poland.1 #Last Will and Testament of [Root.Monarch.GetName]
	on_end = fragmentation_of_poland.100	#End of the Fragmentation Period
	
	on_monthly = {
		events = {
		}
		random_events = {
			1000 = 0
			#All triggered outside of random
			#100 = fragmentation_of_poland.2 #triggered
			#100 = fragmentation_of_poland.3 #Sale of Lubusz
			#100 = fragmentation_of_poland.4 #Pomeranian Ties Broken
			#100 = fragmentation_of_poland.5 #Acts of Cienia
			#100 = fragmentation_of_poland.6 #Magdeburg Laws
			#100 = fragmentation_of_poland.7 #Support from the Church
			#100 = fragmentation_of_poland.8 #Depopulation Crisis
			#100 = fragmentation_of_poland.9 #Halychian Raid
		}
	}
